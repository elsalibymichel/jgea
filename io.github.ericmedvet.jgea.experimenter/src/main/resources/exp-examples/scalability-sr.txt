$treeDesc2 = ea.s.me.d.descriptor(f = f.size(of = f.filter(of = ea.f.treeLeaves(of = ea.f.genotype()); condition = ea.predicate.isSrVariable())); min = 1; max = 32; nOfBins = 8)

  runs = (nOfThreads = [1; 2; 4; 8]) * (randomGenerator = (seed = [1:1:3]) * [m.defaultRG()]) *
    (problem = [
      ea.p.moToSo(objective = mse; moProblem = ea.p.ur.synthetic(name = "nguyen7"));

    (solver = (stop = ea.sc.nOfQualityEvaluations(n = 1000)) * [
      ea.s.asyncScheduledMfMapElites(

    ea.l.console(
      runFunctions = [ea.f.runKey(key = "run.nOfThreads")];

      );
      type = svg;
      configuration = ea.plot.conf.image(
        axesShow = all;
        axesIndependences = [all];
        linesMarkers = true
      );
      overwrite = true
    )
  ]
